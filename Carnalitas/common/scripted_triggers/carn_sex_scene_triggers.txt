#carn_had_sex_with_effect triggers

carn_sex_scene_no_pregnancy_trigger = {
	OR = {
		is_target_in_variable_list = {
			name = carn_sex_scene_option_flag_list
			target = flag:no_pregnancy
		}
		exists = scope:carn_sex_no_pregnancy
	}
}
