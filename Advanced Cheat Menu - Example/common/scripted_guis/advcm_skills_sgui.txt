advcm_add_diplomacy = {
	scope = character
	
	effect = {
		every_living_character = {
			limit = { has_character_flag = marked_for_edit }

			hidden_effect = {
				add_diplomacy_skill = root.var:advcm_diplomacy_value
			}
		}
	}
}

advcm_remove_diplomacy = {
	scope = character
	
	effect = {
		every_living_character = {
			limit = { has_character_flag = marked_for_edit }

			hidden_effect = {
				add_diplomacy_skill = {
					value = root.var:advcm_diplomacy_value
					multiply = -1
				}
			}
		}
	}
}
